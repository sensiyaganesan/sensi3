class Solution
 {
    public TreeNode sortedArrayToBST(int[] nums)
     {
        if (nums == null || nums.length == 0)
         {
            return null;
        }
        return value(nums, 0, nums.length - 1);
    }

    private TreeNode value(int[] nums, int left, int right)
     {
        if (left > right)
         {
            return null;
        }

        int mid = left + (right - left) / 2; 
        TreeNode node = new TreeNode(nums[mid]); 

        node.left = value(nums, left, mid - 1); 
        node.right = value(nums, mid + 1, right);

        return node;
    }
}
